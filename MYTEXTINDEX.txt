    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, onSnapshot, query, orderBy } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = {
            apiKey: "YOUR_API_KEY",
            authDomain: "YOUR_PROJECT.firebaseapp.com",
            projectId: "YOUR_PROJECT_ID",
            storageBucket: "YOUR_BUCKET.appspot.com",
            messagingSenderId: "YOUR_SENDER_ID",
            appId: "YOUR_APP_ID"
        };

        const app = initializeApp(firebaseConfig);
        const auth = getAuth();
        const db = getFirestore();

        // Anonymous login
        signInAnonymously(auth).catch(console.error);

        // Registration form submission
        const form = document.getElementById('registrationForm');
        const regStatus = document.getElementById('regMessageStatus');
        form.addEventListener('submit', async e => {
            e.preventDefault();
            const name = form.regName.value.trim();
            const email = form.regEmail.value.trim();
            const phone = form.regPhone.value.trim();
            const message = form.regMessage.value.trim();
            try {
                await addDoc(collection(db, "yaksha_registrations"), { name, email, phone, message, timestamp: new Date() });
                regStatus.textContent = "ನೋಂದಣಿ ಯಶಸ್ವಿಯಾಗಿದೆ! ಧನ್ಯವಾದಗಳು.";
                form.reset();
            } catch (err) {
                regStatus.textContent = "ದೋಷ! ದಯವಿಟ್ಟು ಪುನಃ ಪ್ರಯತ್ನಿಸಿ.";
                console.error(err);
            }
        });

        // Events rendering
        const eventsList = document.getElementById('eventsList');
        const q = query(collection(db, "yaksha_events"), orderBy("timestamp", "desc"));
        onSnapshot(q, snapshot => {
            eventsList.innerHTML = '';
            if (snapshot.empty) {
                eventsList.innerHTML = '<li class="p-4 bg-yaksha-light-gold rounded-xl shadow-inner text-gray-700 text-center">ಇದೇನೋ ಕಾರ್ಯಕ್ರಮಗಳು ಲಭ್ಯವಿಲ್ಲ</li>';
            } else {
                snapshot.forEach(doc => {
                    const data = doc.data();
                    const li = document.createElement('li');
                    li.className = "p-4 bg-yaksha-light-gold rounded-xl shadow-inner text-gray-700 text-center";
                    li.textContent = `${data.title || 'ಕಾರ್ಯಕ್ರಮ'}: ${data.date || 'ದಿನಾಂಕ ಲಭ್ಯವಿಲ್ಲ'}`;
                    eventsList.appendChild(li);
                });
            }
        });

        // Test: simulate adding a new event
        document.getElementById('simulateEventBtn').addEventListener('click', async () => {
            try {
                await addDoc(collection(db, "yaksha_events"), {
                    title: "ಪ್ರದರ್ಶನ ದಿನಾಚರಣೆ",
                    date: new Date().toLocaleDateString(),
                    timestamp: new Date()
                });
            } catch(e){ console.error(e);}
        });
    </script>
